Particles 3d
<script type="text/javascript" language="javascript">

var n=40;
var LargeSize=15;


var docHeight=document.documentElement.offsetHeight*0.9;

var pa=new Array(n);
var py=new Array(n);
var op=new Array(n);
for(a=0;a<n;a++){
pa[a]=get_random(360);
py[a]=get_random(docHeight);
op[a]=document.createElement("DIV");
op[a].id="p"+a;
document.body.appendChild(op[a]);
op[a].innerHTML="*";
op[a].style.position="absolute";
op[a].style.color="white";

}
Draw();


function Draw(){
docWidth=document.documentElement.offsetWidth*0.9;
for(a=0;a<n;a++){
if(pa[a]>359){
pa[a]=0
}else{
pa[a]++;
}
op[a].style.fontSize=LargeSize+LargeSize/2*Math.sin(pa[a]* Math.PI/180);

MoveElementTo(op[a],1.1*docWidth/2+docWidth/2*Math.cos(pa[a]* Math.PI/180),py[a]);
}
t=setTimeout("Draw()",100)
}

function get_random(n)
{
var ranNum= Math.floor(Math.random()*n);
return ranNum;
}

function MoveElementTo(o,x,y){
o.style.pixelLeft=x;
o.style.pixelTop=y;
}

</script>
